252 size_t lastIndex = m_components.size() - 1;
253
254 if (index != lastIndex) {
255 m_components[index] = std::move(m_components[lastIndex]);
256 uint32_t lastEntity = m_indexToEntity[lastIndex];

264 m_indexToEntity.pop_back();
265 m_entityToIndex.erase(entity); // erase pair entity key, index vaue
